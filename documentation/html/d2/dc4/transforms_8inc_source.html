72 // based on original Reorient() macro by John VanSickle
73 #macro Reorient_Trans(Axis1, Axis2)
74  #local vX1 = vnormalize(Axis1);
75  #local vX2 = vnormalize(Axis2);
76  #local Y = vcross(vX1, vX2);
77  #if(vlength(Y) > 0)
78  #local vY = vnormalize(Y);
79  #local vZ1 = vnormalize(vcross(vX1, vY));
80  #local vZ2 = vnormalize(vcross(vX2, vY));
81  transform {
82  matrix < vX1.x, vY.x,vZ1.x, vX1.y,vY.y,vZ1.y, vX1.z,vY.z, vZ1.z, 0,0,0 >
83  matrix < vX2.x,vX2.y,vX2.z, vY.x,vY.y, vY.z, vZ2.x,vZ2.y,vZ2.z, 0,0,0 >
84  }
85  #else
86  #if (vlength(vX1-vX2)=0)
87  transform {}
88  #else
89  #local vZ = VPerp_To_Vector(vX2);
90  transform { Axis_Rotate_Trans(vZ,180) }
91  #end
92  #end
93 #end

104 // Calculates a transformation which will center the bounding box of Object
105 // along specified axis Axis
106 // Usage:
107 // object {MyObj
108 // Center_Trans(MyObj, x) center along x axis
109 // You can also center along multiple axis:
110 // Center_Trans(MyObj, x+y) center along x and y axis
111 #macro Center_Trans(Object, Axis)
112  #local Mn = min_extent(Object);
113  #local Mx = max_extent(Object);
114  transform {translate -Axis*((Mx - Mn)/2 + Mn)}
115 #end

117 // Calculates a transformation which will align the bounding box
118 // of an object to a point. Negative values on Axis will align to
119 // the sides facing the negative ends of the coordinate system,
120 // positive values will align to the opposite sides, 0 means
121 // not to do any alignment on that axis.
122 // Usage:
123 // object {MyObj
124 // Align_Trans(MyObj, x, Pt)
125 // Align right side of object to be coplanar with Pt
126 // Align_Trans(MyObj, -y, Pt)
127 // Align bottom of object to be coplanar with Pt
128 #macro Align_Trans(Object, Axis, Pt)
129  #local Mn = min_extent(Object);
130  #local Mx = max_extent(Object);
131  transform {
132  #if(Axis.x < -0.5)
133  translate x*(Pt - Mn.x)
134  #else
135  #if(Axis.x > 0.5)
136  translate x*(Pt - Mx.x)
137  #end
138  #end
139 
140  #if(Axis.y < -0.5)
141  translate y*(Pt - Mn.y)
142  #else
143  #if(Axis.y > 0.5)
144  translate y*(Pt - Mx.y)
145  #end
146  #end
147 
148  #if(Axis.z < -0.5)
149  translate z*(Pt - Mn.z)
150  #else
151  #if(Axis.z > 0.5)
152  translate z*(Pt - Mx.z)
153  #end
154  #end
155  }
156 #end

158 // Aligns an object to a spline for a given time value.
159 // The Z axis of the object will point in the forward direction
160 // of the spline and the Y axis of the object will point upwards.
161 //
162 // usage: object {MyObj Spline_Trans(MySpline, clock, y, 0.1, 0.5)}
163 //
164 // Spline: The spline that the object is aligned to.
165 //
166 // Time: The time value to feed to the spline, for example clock.
167 //
168 // Sky: The vector that is upwards in your scene, usually y.
169 //
170 // Foresight: How much in advance the object will turn and bank.
171 // Values close to 0 will give precise results, while higher
172 // values give smoother results. It will not affect parsing
173 // speed, so just find the value that looks best.
174 //
175 // Banking: How much the object tilts when turning. Note that the amount
176 // of tilting is equally much controlled by the ForeSight value.
177 //
178 // Author: Rune S. Johansen
179 #macro Spline_Trans (Spline, Time, Sky, Foresight, Banking)
180  #local Location = <0,0,0>+Spline(Time);
181  #local LocationNext = <0,0,0>+Spline(Time+Foresight);
182  #local LocationPrev = <0,0,0>+Spline(Time-Foresight);
183  #local Forward = vnormalize(LocationNext-Location);
184  #local Right = VPerp_To_Plane(Sky,Forward);
185  #local Up = VPerp_To_Plane(Forward,Right);
186  #local Matrix = Matrix_Trans(Right,Up,Forward,Location)
187  #local BankingRotation =
188  degrees(atan2(
189  VRotation(
190  VProject_Plane((LocationNext-Location),Sky),
191  VProject_Plane((Location-LocationPrev),Sky),
192  Up
193  )*Banking
194  ,1
195  ));
196  transform {
197  rotate BankingRotation*z
198  transform Matrix
199  }
200 #end
