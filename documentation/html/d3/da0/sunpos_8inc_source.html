1 // Persistence of Vision Ray Tracer Scene Include File
2 // File: sunpos.inc
3 // Desc: This macro returns the suns position,
4 // on a given date, time, and location on earth.
5 // Date: 15-09-1998
6 // Updated : 2001-07-27
7 // Auth: Ingo Janssen
8 //
9 // ======================================================
10 //
11 // Assumption: in the scene north is in the +Z direction, south is -Z.
12 //
13 // Invoke in your scene:
14 // light_source {
15 // SunPos(Year, Month, Day, Hour, Minute, Lstm, LAT, LONG)
16 // rgb 1
17 // }
18 //
19 // Greenwich, noon on the longest day of 2000 (no daylight saving):
20 // SunPos(2000, 6, 21, 12, 2, 0, 51.4667, 0.00)
21 //
22 // Year: in four digits
23 // Month: number
24 // Dat: number
25 // Time: in 24 hour format
26 // Lstm: Meridian of your local timezone (+1 hour = +15 deg)
27 // in degrees (east = positive, west = negative)
28 // Location on earth:
29 // LAT: Lattitude in degrees.decimal, northern hemisphere = positive, southern = negative
30 // LONG: Longitude in degrees.decimal, east = positive, west is negative
31 //
32 // The macro returns the position of the sun, but also declares it as the vector SolarPosition.
33 // Also the found Azimuth (Az) and Altitude (Al) are declared, this can be usefull for
34 // aligning an object with the sunlight: cylinder{
35 // <-2,0,0>,<2,0,0>,0.1
36 // rotate <0, Az-90, Al>
37 // texture {.. LightRay ..}
38 // }
39 //
40 // ======================================================
41 //
42 // Find your local position at http://gnpswww.nima.mil/geonames/GNS/ (Gazetteer search)
43 //
44 // Local time(zone) at http://www.hilink.com.au/times/
45 //
46 // Equations used here can be found at http://hotel04.ausys.se/pausch/english.htm (Computing rise/set times)
47 // not only for the sun but also for the moon, planets and other stuff up there.
48 //
49 // ======================================================
50 //
51 
52 #macro SunPos(Year, Month, Day, Hour, Minute, Lstm, LAT, LONG)
53 
54  #if (abs(LONG-Lstm)>30)
55  #debug "\nREMARK: \nLongitude differs by more than 30 degrees from timezone meridian.\n"
56  #debug concat("Local timezone meridian is:",str(Lstm,5,0),"\n")
57  #debug concat("Longitude is:",str(LONG,5,0),"\n")
58  #end
59 
60  // Calculate universal time (UT)
61  #local T= Hour+(Minute/60);
62  #local UT= T-Lstm/15;
63  #if (0>UT)
64  #local Day= Day-1;
65  #local UT= 24+UT;
66  #end
67  #if (UT>24)
68  #local Day= Day+1;
69  #local UT= UT-24;
70  #end
71 
72  // Amount of days to, or from, the year 2000
73  #local d= 367*Year-int((7*int((Year+int((Month+9))/12)))/4)+int((275*Month)/9)+Day-730530+UT/24;
74 
75  // Longitude of perihelion (w), eccentricity (e)
76  #local w= 282.9404+4.70935E-5*d;
77  #local e= 0.016709-1.151E-9*d;
78 
79  // Mean anomaly (M), sun's mean longitude (L)
80  #local M= 356.0470+0.9856002585*d;
81  #if (0<M<360)
82  #local M= M-floor(M/360)*360;
83  #end
84  #local L= w+M;
85  #if (0<L<360)
86  #local L= L-floor(L/360)*360;
87  #end
88 
89  // Obliquity of the ecliptic, eccentric anomaly (E)
90  #local oblecl= 23.4393-3.563E-7*d;
91  #local E= M+(180/pi)*e*sin(radians(M))*(1+e*cos(radians(M)));
92 
93  // Sun's rectangular coordinates in the plane of ecliptic (A,B)
94  #local A= cos(radians(E))-e;
95  #local B= sin(radians(E))*sqrt(1-e*e);
96 
97  // Distance (r), true anomaly (V), longitude of the sun (lon)
98  #local r= sqrt(A*A+B*B);
99  #local V= degrees(atan2(radians(B),radians(A)));
100  #local lon= V+w;
101  #if (0<lon<360)
102  #local lon= lon-floor(lon/360)*360;
103  #end
104 
105  // Calculate declination (Decl) and right ascension (RA)
106  #local Decl= degrees(asin(sin(radians(oblecl))*sin(radians(lon))));
107  #local RA= degrees(atan2(sin(radians(lon))*cos(radians(oblecl)),cos(radians(lon))))/15;
108 
109  // Greenwich meridian siderial time at 00:00 (GMST0),siderial time (SIDTIME), hour angle (HA)
110  #local GMST0= L/15+12;
111  #local SIDTIME= GMST0+UT+LONG/15;
112  #local HA= (SIDTIME-RA)*15;
113 
114  // This is what we're looking for: Altitude & Azimuth
115  #declare Al= degrees(asin(sin(radians(LAT))*sin(radians(Decl))+cos(radians(LAT))*cos(radians(Decl))*cos(radians(HA))));
116  #declare Az= degrees(atan2(sin(radians(HA)),cos(radians(HA))*sin(radians(LAT))-tan(radians(Decl))*cos(radians(LAT))))+180;
117 
118  #declare SolarPosition=vrotate(<0,0,1000000000>,<-Al,Az,0>);
119  (SolarPosition)
120 
121 #end
