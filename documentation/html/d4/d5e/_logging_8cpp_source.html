31 void RLBinUtils::RLBin_Multi(std::string str, LogType _log_type)
32 {
33  if(_log_type == T_DEBUG) RLBin_Debug(str, __FILENAME__, __LINE__);
34  else if (_log_type == T_LOG) RLBin_Log(str, __FILENAME__);
35  else if (_log_type == T_ERROR) RLBin_Error(str, __FILENAME__, __LINE__);
36  else if (_log_type == T_DIS) RLBin_Dis(str);
37  else if (_log_type == T_TRAM) RLBin_Tram(str);
38  else if (_log_type == T_LIBCALL) RLBin_LibCalls(str);
39  else if (_log_type == T_OPTSTAT) RLBin_OptStat(str);
40  else if (_log_type == T_TIMESTAT) RLBin_TimeStat(str);
41  else if (_log_type == T_MODLOG) RLBin_ModLog(str);
42  else if (_log_type == T_THREADLOG) RLBin_ThreadLog(str);
43  else if (_log_type == T_STATIC) RLBin_Static(str);
44  else RLBin_Error("Unexpected Log Type\n", __FILENAME__, __LINE__);
45 }

47 void RLBinUtils::RLBin_Error(std::string errStr, std::string source_file, int source_file_line)
48 {
49  // Get the last error code as string
50  std::string error_code = std::string();
51  DWORD error = GetLastError();
52  if (error)
53  {
54  LPVOID lpMsgBuf;
55  DWORD bufLen = FormatMessage(
56  FORMAT_MESSAGE_ALLOCATE_BUFFER |
57  FORMAT_MESSAGE_FROM_SYSTEM |
58  FORMAT_MESSAGE_IGNORE_INSERTS,
59  NULL,
60  error,
61  MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
62  (LPTSTR) &lpMsgBuf,
63  0, NULL );
64  if (bufLen)
65  {
66  LPCSTR lpMsgStr = (LPCSTR)lpMsgBuf;
67  std::string result(lpMsgStr, lpMsgStr+bufLen);
68 
69  LocalFree(lpMsgBuf);
70 
71  error_code = result;
72  }
73  }
74 
75  // Now print to file
76  printf("RL-Bin has encountered some error and will exit! \nPlease check the error log for details.\n");
77  if (!errorFile)
78  {
79  string fileName = string(BASE_FOLDER) + ConvertIntToString(Analysis_Number) + string("\\") + BASE_ERROR_FILENAME + string(".txt");
80  errorFile = fopen(fileName.c_str(),"a+");
81  }
82  if (errorFile)
83  {
84  std::string tempStr= GetTime();
85  tempStr.append("\t");
86  tempStr.resize(20, ' ');
87  tempStr.append(source_file);
88  tempStr.resize(50, ' ');
89  tempStr.append("\t");
90  tempStr.append(ConvertIntToString(source_file_line));
91  tempStr.append("\t");
92  tempStr.resize(60, ' ');
93  tempStr.append(errStr);
94  tempStr.append("\nError Code:\t");
95  tempStr.append(error_code);
96  fprintf(errorFile, "%s\n", tempStr.c_str());
97  fflush(errorFile);
98  }
99 
100  abort();
101 }
