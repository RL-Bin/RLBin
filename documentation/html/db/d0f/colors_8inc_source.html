163 // Takes Hue value as input, returns RGB vector.
164 #macro CH2RGB (H)
165  #local H = mod(H, 360);
166  #local H = (H < 0 ? H+360 : H);
167  #switch (H)
168  #range (0, 120)
169  #local R = (120- H) / 60;
170  #local G = ( H- 0) / 60;
171  #local B = 0;
172  #break
173  #range (120, 240)
174  #local R = 0;
175  #local G = (240- H) / 60;
176  #local B = ( H-120) / 60;
177  #break
178  #range (240, 360)
179  #local R = ( H-240) / 60;
180  #local G = 0;
181  #local B = (360- H) / 60;
182  #break
183  #end
184  <min(R,1), min(G,1), min(B,1)>
185 #end

189 #macro CRGB2H (RGB, Max, Span)
190  #local H = 0;
191  #local R = RGB.red;

193  #local B = RGB.blue;

198  + (B = Max & R != Max ? 4 + (R - G)/Span : 0)
199  )*60;
200  #end
201  H
202 #end

204 // Converts a color in HSL color space to a color in RGB color space.
205 // Input: < Hue, Saturation, Lightness, Filter, Transmit >
206 // Output: < Red, Green, Blue, Filter, Transmit >
207 #macro CHSL2RGB(Color)
208  #local HSLFT = color Color;
209  #local H = (HSLFT.red);
210  #local S = (HSLFT.green);
211  #local L = (HSLFT.blue);
212  #local SatRGB = CH2RGB(H);
213  #local Col = 2*S*SatRGB + (1-S)*<1,1,1>;
214  #if (L<0.5)
215  #local RGB = L*Col;
216  #else
217  #local RGB = (1-L)*Col + (2*L-1)*<1,1,1>;
218  #end
219  <RGB.red,RGB.green,RGB.blue,(HSLFT.filter),(HSLFT.transmit)>
220 #end

222 // Converts a color in RGB color space to a color in HSL color space.
223 // Input: < Red, Green, Blue, Filter, Transmit >
224 // Output: < Hue, Saturation, Lightness, Filter, Transmit >
225 #macro CRGB2HSL(Color)
226  #local RGBFT = color Color;
227  #local R = (RGBFT.red);
228  #local G = (RGBFT.green);
229  #local B = (RGBFT.blue);
230  #local Min = min(R,min(G,B));
231  #local Max = max(R,max(G,B));
232  #local Span = Max-Min;
233  #local L = (Min+Max)/2;
234  #local S = 0;
235  #if( L!=0 & L!=1 )
236  #local S = Span / ( L<0.5 ? (L*2) : (2-L*2) );
237  #end
238  #local H = CRGB2H (<R,G,B>, Max, Span);
239  <H,S,L,(RGBFT.filter),(RGBFT.transmit)>
240 #end
